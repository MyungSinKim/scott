Database생성, User 생성

CREATE DATABASE scott;
CREATE USER 'scott'@'localhost' IDENTIFIED BY 'scott';
GRANT ALL PRIVILEGES ON *.* TO 'scott'@'localhost'
    WITH GRANT OPTION;
CREATE USER 'scott'@'%' IDENTIFIED BY 'scott';
GRANT ALL PRIVILEGES ON *.* TO 'scott'@'%'
     WITH GRANT OPTION;
FLUSH PRIVILEGES;
